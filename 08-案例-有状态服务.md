# 08 - 案例：有状态服务（MySQL 主从复制）

本案例演示如何使用 StatefulSet 部署 MySQL 主从复制集群，涵盖有状态应用的核心模式：稳定标识、独立存储、有序部署。

## 架构设计

```
           ┌──────────────────────────────────────────┐
           │           MySQL Cluster                   │
           │                                          │
  写入 ──► │  mysql-0 (Master)                        │
           │    ├── PVC: data-mysql-0 (10Gi)           │
           │    └── 复制 binlog →                      │
           │                                          │
  读取 ◄── │  mysql-1 (Slave)                         │
           │    ├── PVC: data-mysql-1 (10Gi)           │
           │    └── ← 从 Master 复制                  │
           │                                          │
  读取 ◄── │  mysql-2 (Slave)                         │
           │    ├── PVC: data-mysql-2 (10Gi)           │
           │    └── ← 从 Master 复制                  │
           │                                          │
           └──────────────────────────────────────────┘
                     │
           Headless Service: mysql-headless
           DNS: mysql-{0,1,2}.mysql-headless.webapp.svc.cluster.local
```

**核心需求：**
- Master（mysql-0）接受写入，Slave（mysql-1, mysql-2）只读
- 每个实例独立 PVC，Pod 重启后数据不丢失
- 稳定 DNS 标识，Pod 重启后身份不变

## 第一步：配置与 Secret

```yaml
# 01-config.yaml
apiVersion: v1
kind: Secret
metadata:
  name: mysql-secret
  namespace: webapp
type: Opaque
stringData:
  ROOT_PASSWORD: "RootP@ss2024"
  REPLICATION_PASSWORD: "ReplP@ss2024"
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: mysql-config
  namespace: webapp
data:
  # Master 配置（mysql-0 使用）
  master.cnf: |
    [mysqld]
    server-id=1
    log-bin=mysql-bin
    binlog-format=ROW
    gtid-mode=ON
    enforce-gtid-consistency=ON
    sync-binlog=1
    innodb-flush-log-at-trx-commit=1

  # Slave 配置（mysql-1, mysql-2 使用）
  slave.cnf: |
    [mysqld]
    server-id=2
    relay-log=mysql-relay-bin
    read-only=ON
    gtid-mode=ON
    enforce-gtid-consistency=ON
    log-slave-updates=ON

  # 初始化脚本
  init.sql: |
    -- 创建复制用户
    CREATE USER IF NOT EXISTS 'repl'@'%' IDENTIFIED BY 'ReplP@ss2024';
    GRANT REPLICATION SLAVE ON *.* TO 'repl'@'%';

    -- 创建应用用户
    CREATE DATABASE IF NOT EXISTS appdb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
    CREATE USER IF NOT EXISTS 'app'@'%' IDENTIFIED BY 'AppP@ss2024';
    GRANT ALL PRIVILEGES ON appdb.* TO 'app'@'%';
    FLUSH PRIVILEGES;
```

## 第二步：Headless Service

```yaml
# 02-service.yaml
# Headless Service - 为每个 Pod 提供独立 DNS
apiVersion: v1
kind: Service
metadata:
  name: mysql-headless
  namespace: webapp
spec:
  clusterIP: None              # Headless
  selector:
    app: mysql
  ports:
  - port: 3306
    targetPort: 3306
    name: mysql
---
# 普通 Service - 供应用读写访问（指向 Master）
apiVersion: v1
kind: Service
metadata:
  name: mysql-rw
  namespace: webapp
spec:
  selector:
    app: mysql
    role: master               # 只匹配 Master Pod
  ports:
  - port: 3306
    targetPort: 3306
---
# 普通 Service - 供应用只读访问（指向所有 Slave）
apiVersion: v1
kind: Service
metadata:
  name: mysql-ro
  namespace: webapp
spec:
  selector:
    app: mysql
    role: slave                # 只匹配 Slave Pod
  ports:
  - port: 3306
    targetPort: 3306
```

## 第三步：StatefulSet

```yaml
# 03-statefulset.yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: mysql
  namespace: webapp
spec:
  serviceName: mysql-headless
  replicas: 3
  selector:
    matchLabels:
      app: mysql
  template:
    metadata:
      labels:
        app: mysql
    spec:
      # 初始化容器：根据 Pod 序号选择 Master/Slave 配置
      initContainers:
      - name: init-mysql
        image: mysql:8.0
        command:
        - bash
        - -c
        - |
          # 根据 Pod 序号设置 server-id 和角色
          ordinal=${HOSTNAME##*-}
          echo "[mysqld]" > /mnt/conf.d/server.cnf
          echo "server-id=$((100 + ordinal))" >> /mnt/conf.d/server.cnf

          if [ "$ordinal" -eq 0 ]; then
            # Master 配置
            cp /mnt/config-map/master.cnf /mnt/conf.d/role.cnf
            echo "master" > /mnt/labels/role
          else
            # Slave 配置
            cp /mnt/config-map/slave.cnf /mnt/conf.d/role.cnf
            echo "slave" > /mnt/labels/role
          fi
        volumeMounts:
        - name: conf
          mountPath: /mnt/conf.d
        - name: config-map
          mountPath: /mnt/config-map
        - name: labels
          mountPath: /mnt/labels

      # 克隆数据容器（Slave 启动时从 Master 拉取数据）
      - name: clone-data
        image: mysql:8.0
        command:
        - bash
        - -c
        - |
          ordinal=${HOSTNAME##*-}
          if [ "$ordinal" -eq 0 ] || [ -d /var/lib/mysql/mysql ]; then
            exit 0  # Master 或已有数据，跳过
          fi
          # 从上一个节点克隆数据
          prev_ordinal=$((ordinal - 1))
          source="mysql-${prev_ordinal}.mysql-headless.webapp.svc.cluster.local"
          xtrabackup --backup \
            --host="$source" \
            --user=repl \
            --password="${REPLICATION_PASSWORD}" \
            --target-dir=/var/lib/mysql
        envFrom:
        - secretRef:
            name: mysql-secret
        volumeMounts:
        - name: data
          mountPath: /var/lib/mysql

      containers:
      - name: mysql
        image: mysql:8.0
        ports:
        - containerPort: 3306
          name: mysql
        env:
        - name: MYSQL_ROOT_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mysql-secret
              key: ROOT_PASSWORD
        - name: REPLICATION_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mysql-secret
              key: REPLICATION_PASSWORD
        resources:
          requests:
            cpu: "250m"
            memory: "512Mi"
          limits:
            cpu: "1"
            memory: "1Gi"
        volumeMounts:
        - name: data
          mountPath: /var/lib/mysql
        - name: conf
          mountPath: /etc/mysql/conf.d
        - name: init-scripts
          mountPath: /docker-entrypoint-initdb.d
        - name: labels
          mountPath: /etc/labels
        readinessProbe:
          exec:
            command:
            - bash
            - -c
            - |
              mysql -u root -p"${MYSQL_ROOT_PASSWORD}" -e "SELECT 1"
          initialDelaySeconds: 30
          periodSeconds: 10
        livenessProbe:
          exec:
            command:
            - bash
            - -c
            - |
              mysqladmin -u root -p"${MYSQL_ROOT_PASSWORD}" ping
          initialDelaySeconds: 60
          periodSeconds: 15
          failureThreshold: 5

      # 启动后设置角色标签（供 Service selector 使用）
      lifecycle:
        postStart:
          exec:
            command:
            - bash
            - -c
            - |
              ordinal=${HOSTNAME##*-}
              role=$(cat /etc/labels/role)
              kubectl label pod ${HOSTNAME} -n webapp role=${role} --overwrite

      volumes:
      - name: conf
        emptyDir: {}
      - name: config-map
        configMap:
          name: mysql-config
      - name: init-scripts
        configMap:
          name: mysql-config
          items:
          - key: init.sql
            path: init.sql
      - name: labels
        emptyDir: {}

  # PVC 模板 - 每个 Pod 独立持久化存储
  volumeClaimTemplates:
  - metadata:
      name: data
    spec:
      accessModes: ["ReadWriteOnce"]
      storageClassName: standard
      resources:
        requests:
          storage: 10Gi
```

## 第四步：部署与验证

```bash
# 部署
kubectl apply -f 01-config.yaml
kubectl apply -f 02-service.yaml
kubectl apply -f 03-statefulset.yaml

# 观察有序启动
kubectl get pods -n webapp -w
NAME      READY   STATUS              AGE
mysql-0   0/1     ContainerCreating   5s
mysql-0   1/1     Running             30s     ← mysql-0 Ready 后才开始创建 mysql-1
mysql-1   0/1     ContainerCreating   31s
mysql-1   1/1     Running             60s     ← mysql-1 Ready 后才开始创建 mysql-2
mysql-2   0/1     ContainerCreating   61s
mysql-2   1/1     Running             90s
```

### 验证主从复制

```bash
# 检查 Master 状态
kubectl exec mysql-0 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "SHOW MASTER STATUS\G"

# 检查 Slave 复制状态
kubectl exec mysql-1 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "SHOW SLAVE STATUS\G"

# 测试写入
kubectl exec mysql-0 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "
    CREATE DATABASE testdb;
    USE testdb;
    CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(50));
    INSERT INTO users VALUES (1, 'Alice');
  "

# 验证数据同步到 Slave
kubectl exec mysql-2 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "SELECT * FROM testdb.users"
# +----+-------+
# | id | name  |
# +----+-------+
# |  1 | Alice |
# +----+-------+
```

### 验证读写分离

```bash
# 写入通过 mysql-rw Service（只指向 Master）
kubectl exec mysql-0 -n webapp -- \
  mysql -h mysql-rw -u root -pRootP@ss2024 -e "
    INSERT INTO testdb.users VALUES (2, 'Bob');
  "

# 读取通过 mysql-ro Service（指向 Slave）
kubectl run test-read --image=mysql:8.0 -n webapp --rm -it -- \
  mysql -h mysql-ro -u root -pRootP@ss2024 -e "SELECT * FROM testdb.users"
# +----+-------+
# | id | name  |
# +----+-------+
# |  1 | Alice |
# |  2 | Bob   |
# +----+-------+
```

### 验证存储持久化

```bash
# 查看 PVC
kubectl get pvc -n webapp
NAME             STATUS   VOLUME    CAPACITY   STORAGECLASS
data-mysql-0     Bound    pv-001    10Gi       standard
data-mysql-1     Bound    pv-002    10Gi       standard
data-mysql-2     Bound    pv-003    10Gi       standard

# 删除 Pod（模拟故障）
kubectl delete pod mysql-1 -n webapp

# Pod 自动重建，使用相同 PVC，数据不丢失
kubectl get pods -n webapp -w
mysql-1   0/1     ContainerCreating   0s
mysql-1   1/1     Running             20s

# 验证数据仍在
kubectl exec mysql-1 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "SELECT * FROM testdb.users"
```

## Pod 故障恢复

```bash
# Master 故障时：需要手动切换（生产环境建议使用 Orchestrator 等自动化工具）
# 1. 提升 mysql-1 为新 Master
kubectl exec mysql-1 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "
    STOP SLAVE;
    RESET SLAVE ALL;
    SET GLOBAL read_only=OFF;
  "

# 2. 更新 mysql-1 的角色标签
kubectl label pod mysql-1 -n webapp role=master --overwrite

# 3. 其他 Slave 切换到新 Master
kubectl exec mysql-2 -n webapp -- \
  mysql -u root -pRootP@ss2024 -e "
    STOP SLAVE;
    CHANGE MASTER TO
      MASTER_HOST='mysql-1.mysql-headless.webapp.svc.cluster.local',
      MASTER_USER='repl',
      MASTER_PASSWORD='ReplP@ss2024',
      MASTER_AUTO_POSITION=1;
    START SLAVE;
  "
```

## 知识点回顾

| 使用的资源 | 作用 |
|-----------|------|
| StatefulSet | 有序部署、稳定标识、独立存储 |
| Headless Service | 为每个 Pod 提供独立 DNS |
| 普通 Service × 2 | 读写分离（rw → Master, ro → Slave） |
| PVC (per Pod) | 独立持久化存储 |
| ConfigMap | MySQL 配置文件 |
| Secret | 数据库密码 |
| Init Container | 根据序号选择配置 |
| Readiness Probe | 健康检查，影响 Service 路由 |
